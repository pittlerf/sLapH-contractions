52  BlockIterator operator++() {
53  if (type_ == DilutionType::block) {
54  int const block_size = num_slice_ / num_block_;
55  int const block_source_begin = block_source_ * block_size;
56  int const block_source_end = (block_source_ + 1) * block_size;
57  int const block_sink_begin = block_sink_ * block_size;
58  int const block_sink_end = (block_sink_ + 1) * block_size;
59 
60  // Go to the next sink slice.
61  if (one_sink_slice_) {
62  slice_sink_ += block_size;
63  } else {
64  ++slice_sink_;
65  }
66 
67  // If we iterated through all sinks in this block, we need to go to the
68  // next source.
69  if (slice_sink_ == block_sink_end) {
70  slice_sink_ = block_sink_begin;
71  ++slice_source_;
72  }
73 
74  // If we iterated through all the sources in our block, we need to
75  // exchange
76  // source and sink block and repeat.
77  if (slice_source_ == block_source_end) {
78  slice_source_ = block_sink_begin;
79  slice_sink_ = block_source_begin;
80  ++pass_;
81 
82  if (block_source_ == block_sink_) {
83  ++pass_;
84  }
85 
86  std::swap(block_source_, block_sink_);
87  }
88  } else if (type_ == DilutionType::interlace) {
89  int const block_size = num_slice_ / num_block_;
90  int const block_source_begin = block_source_;
91  int const block_sink_begin = block_sink_;
92 
93  // Go to the next sink slice.
94  if (one_sink_slice_) {
95  slice_sink_ += num_slice_;
96  } else {
97  slice_sink_ += block_size;
98  }
99 
100  // If we iterated through all sinks in this block, we need to go to the
101  // next source.
102  if (slice_sink_ >= num_slice_) {
103  slice_sink_ = block_sink_begin;
104  slice_source_ += block_size;
105  }
106 
107  // If we iterated through all the sources in our block, we need to
108  // exchange
109  // source and sink block and repeat.
110  if (slice_source_ >= num_slice_) {
111  slice_source_ = block_sink_begin;
112  slice_sink_ = block_source_begin;
113  ++pass_;
114 
115  if (block_source_ == block_sink_) {
116  ++pass_;
117  }
118 
119  std::swap(block_source_, block_sink_);
120  }
121  } else {
122  throw std::domain_error("This dilution scheme is not implemented.");
123  }
124 
125  return *this;
126  }
